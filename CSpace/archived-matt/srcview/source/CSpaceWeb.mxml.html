    private function checkLogin(evt:ResultEvent):void
    {
        if(evt.result.loginsuccess == "yes")
        {
            currentState = "Logged In";
        }

        if(evt.result.loginsuccess == "no")
        {
            mx.controls.Alert.show('Invalid username/password','Login Problem');
        }        
    }
